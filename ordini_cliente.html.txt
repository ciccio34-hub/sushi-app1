${p.nome} -

<span class="${p.stato}">
${p.stato}
</span>

<button onclick="pronto('${docSnap.id}',${index})">

</button>

</div>
`;

});

<button onclick="elimina('${docSnap.id}')">

</button>
`;

});

});

window.pronto = async function(id,index){

const ref = doc(db,"ordini",id);

const snap = await getDoc(ref);

const data = snap.data();

data.piatti[index].stato = "pronto";

await updateDoc(ref,{
piatti:data.piatti
});

}

window.elimina = async function(id){

await deleteDoc(doc(db,"ordini",id));

}
