const app = initializeApp(firebaseConfig);
const db = getFirestore(app);

let cart=[];

window.add=function(name,price){
let found=cart.find(i=>i.name===name);
if(!found){ cart.push({name,price,qty:1}); }
else{ found.qty++; }
updateCart();
};

function updateCart(){
let div=document.getElementById("cartItems");
let total=document.getElementById("total");
div.innerHTML="";
let sum=0;
cart.forEach(i=>{
sum+=i.price*i.qty;
div.innerHTML+=i.name+" x"+i.qty+"<br>";
});
total.innerText=sum;
}

window.sendOrder=async function(){
if(cart.length===0){ alert("Carrello vuoto"); return; }
await addDoc(collection(db,"orders"),{
items:cart,
stato:"In preparazione",
time:new Date()
});
cart=[];
updateCart();
};

onSnapshot(collection(db,"orders"),snapshot=>{
let div=document.getElementById("orders");
div.innerHTML="";
snapshot.forEach(doc=>{
let data=doc.data();
let color=data.stato==="Pronto"?"lime":"red";
let text="";
data.items.forEach(i=>{
text+=i.name+" x"+i.qty+"<br>";
});
div.innerHTML+=`
<div class="orderBox">
${text}
Stato: <span style="color:${color}">${data.stato}</span>
</div>
`;
});
});
