<script type="module">

import { db, collection, addDoc } from "./firebase.js";

async function sendOrder(){

const order = {
table: "Tavolo 3",
items:[
{name:"Salmon Nigiri",qty:2},
{name:"Tuna Roll",qty:1}
],
time:new Date()
};

await addDoc(collection(db,"orders"),order);

alert("Ordine inviato!");

}

window.sendOrder = sendOrder;
